import java.util.Locale;
import java.util.Scanner;

public class Main {

		double peso, altura, imc;

		System.out.printf("O IMC Ã© %.2f%n", imc);

			}
			else if (imc <= 31.1) {
				System.out.println("Acima do peso ideal");
			} else {

			}
		}
